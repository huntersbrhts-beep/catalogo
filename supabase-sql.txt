-- 1) Colunas extras na tabela produtos
alter table produtos add column if not exists estoque integer;
alter table produtos add column if not exists ativo boolean default true;
alter table produtos add column if not exists tamanhos jsonb;

-- 2) Tabela de pedidos
create table if not exists pedidos (
  id uuid primary key default gen_random_uuid(),
  created_at timestamptz default now(),
  cliente_nome text,
  telefone text,
  tipo_pedido text,
  bairro text,
  endereco text,
  itens jsonb,
  subtotal numeric default 0,
  taxa_entrega numeric default 0,
  desconto numeric default 0,
  total_final numeric default 0,
  status text default 'novo'
);

-- 3) Tabela de configurações da loja
-- Aqui salvam: cupons, limite da roleta, prêmios da roleta, frete por bairro,
-- banner, horário, redes sociais, aparência e ordem das categorias.
create table if not exists loja_config (
  chave text primary key,
  valor jsonb not null,
  updated_at timestamptz default now()
);

-- 5) Políticas simples para projeto público GitHub Pages
alter table pedidos enable row level security;
alter table loja_config enable row level security;

drop policy if exists "pedidos_insert_publico" on pedidos;
create policy "pedidos_insert_publico" on pedidos for insert with check (true);

drop policy if exists "pedidos_select_publico" on pedidos;
create policy "pedidos_select_publico" on pedidos for select using (true);

drop policy if exists "pedidos_update_publico" on pedidos;
create policy "pedidos_update_publico" on pedidos for update using (true) with check (true);

drop policy if exists "loja_config_select_publico" on loja_config;
create policy "loja_config_select_publico" on loja_config for select using (true);

drop policy if exists "loja_config_insert_publico" on loja_config;
create policy "loja_config_insert_publico" on loja_config for insert with check (true);

drop policy if exists "loja_config_update_publico" on loja_config;
create policy "loja_config_update_publico" on loja_config for update using (true) with check (true);
