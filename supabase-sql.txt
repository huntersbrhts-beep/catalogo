-- Rode este arquivo no Supabase SQL Editor.
-- A roleta NÃO usa mais config_roleta. Agora usa roleta_premios e roleta_config.

create extension if not exists pgcrypto;

create table if not exists public.roleta_config (
  id integer primary key default 1,
  limite integer not null default 0,
  updated_at timestamptz not null default now(),
  constraint roleta_config_id_um check (id = 1)
);

create table if not exists public.roleta_premios (
  id uuid primary key default gen_random_uuid(),
  texto text not null,
  tipo text not null default 'nenhum' check (tipo in ('percentual','valor','nenhum')),
  valor numeric not null default 0,
  ordem integer not null default 1,
  ativo boolean not null default true,
  created_at timestamptz not null default now(),
  updated_at timestamptz not null default now()
);

alter table public.roleta_config enable row level security;
alter table public.roleta_premios enable row level security;

drop policy if exists "roleta_config_read_all" on public.roleta_config;
drop policy if exists "roleta_config_write_authenticated" on public.roleta_config;
drop policy if exists "roleta_premios_read_all" on public.roleta_premios;
drop policy if exists "roleta_premios_insert_authenticated" on public.roleta_premios;
drop policy if exists "roleta_premios_update_authenticated" on public.roleta_premios;
drop policy if exists "roleta_premios_delete_authenticated" on public.roleta_premios;

create policy "roleta_config_read_all"
on public.roleta_config
for select
to anon, authenticated
using (true);

create policy "roleta_config_write_authenticated"
on public.roleta_config
for all
to authenticated
using (true)
with check (true);

create policy "roleta_premios_read_all"
on public.roleta_premios
for select
to anon, authenticated
using (true);

create policy "roleta_premios_insert_authenticated"
on public.roleta_premios
for insert
to authenticated
with check (true);

create policy "roleta_premios_update_authenticated"
on public.roleta_premios
for update
to authenticated
using (true)
with check (true);

create policy "roleta_premios_delete_authenticated"
on public.roleta_premios
for delete
to authenticated
using (true);

insert into public.roleta_config (id, limite, updated_at)
values (1, 30, now())
on conflict (id) do nothing;

-- Só cria exemplos se a tabela estiver vazia. Depois você altera pelo ADM.
insert into public.roleta_premios (texto, tipo, valor, ordem, ativo)
select * from (values
  ('5% OFF','percentual',5,1,true),
  ('10% OFF','percentual',10,2,true),
  ('R$ 5 OFF','valor',5,3,true),
  ('SEM CUPOM','nenhum',0,4,true)
) as v(texto,tipo,valor,ordem,ativo)
where not exists (select 1 from public.roleta_premios);
