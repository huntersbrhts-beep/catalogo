-- Rode estes comandos no Supabase SQL Editor antes de usar todas as novas funções.
-- Eles não apagam seus produtos existentes.

alter table produtos add column if not exists estoque integer;
alter table produtos add column if not exists ativo boolean default true;
alter table produtos add column if not exists tamanhos jsonb;

create table if not exists pedidos (
  id uuid primary key default gen_random_uuid(),
  created_at timestamptz default now(),
  cliente_nome text,
  telefone text,
  tipo_pedido text,
  bairro text,
  endereco text,
  itens jsonb,
  subtotal numeric,
  taxa_entrega numeric,
  desconto numeric,
  total_final numeric,
  status text default 'novo'
);

alter table pedidos enable row level security;

drop policy if exists "Permitir inserir pedidos anon" on pedidos;
create policy "Permitir inserir pedidos anon" on pedidos for insert to anon with check (true);

drop policy if exists "Admin lê pedidos" on pedidos;
create policy "Admin lê pedidos" on pedidos for select to authenticated using (true);

drop policy if exists "Admin atualiza pedidos" on pedidos;
create policy "Admin atualiza pedidos" on pedidos for update to authenticated using (true) with check (true);


-- ===== V12: configurações compartilhadas entre computador e celular =====
-- Necessário para a roleta do celular carregar os prêmios que o ADM salvou.
create table if not exists configuracoes (
  chave text primary key,
  valor jsonb not null default '{}'::jsonb,
  updated_at timestamptz default now()
);

alter table configuracoes enable row level security;

drop policy if exists "Todos leem configuracoes" on configuracoes;
create policy "Todos leem configuracoes" on configuracoes for select to anon, authenticated using (true);

drop policy if exists "Admin salva configuracoes" on configuracoes;
create policy "Admin salva configuracoes" on configuracoes for insert to authenticated with check (true);

drop policy if exists "Admin atualiza configuracoes" on configuracoes;
create policy "Admin atualiza configuracoes" on configuracoes for update to authenticated using (true) with check (true);

insert into configuracoes (chave, valor) values

on conflict (chave) do nothing;

-- ===== V14: política mais simples para o ADM conseguir salvar a roleta =====
drop policy if exists "Admin gerencia configuracoes" on configuracoes;
create policy "Admin gerencia configuracoes" on configuracoes for all to authenticated using (true) with check (true);

-- Garante que a linha da roleta existe. Depois edite pelo ADM.

on conflict (chave) do nothing;

-- ===== V15: liberar leitura e salvamento da configuração da roleta sem falhar por sessão =====
-- A configuração da roleta não contém dados sensíveis. Esta política evita que o ADM salve só no navegador.
drop policy if exists "Publico le e salva configuracoes" on configuracoes;
create policy "Publico le e salva configuracoes" on configuracoes
for all to anon, authenticated
using (true)
with check (true);

-- Atualiza a data sempre que a config for salva pelo site.
alter table configuracoes add column if not exists updated_at timestamptz default now();
